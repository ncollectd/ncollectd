plugin_option(PLUGIN_CERTMONGER "Monitor certmonger request" ON)

set(BUILD_PLUGIN_CERTMONGER 0)

if(PLUGIN_CERTMONGER)
    find_package(LibSystemd)
    if (LIBSYSTEMD_FOUND)
        check_include_file(systemd/sd-bus.h HAVE_SYSTEMD_SD_BUS_H)
        if (HAVE_SYSTEMD_SD_BUS_H)
            set(BUILD_PLUGIN_CERTMONGER 1)
        else()
            set(BUILD_PLUGIN_CERTMONGER_REASON "systemd/sd-bus.h not found" PARENT_SCOPE)
        endif()
    else()
        set(BUILD_PLUGIN_CERTMONGER_REASON "libsystemd not found" PARENT_SCOPE)
    endif()
endif()

set(BUILD_PLUGIN_CERTMONGER ${BUILD_PLUGIN_CERTMONGER} PARENT_SCOPE)

if(BUILD_PLUGIN_CERTMONGER)
    set(PLUGIN_CERTMONGER_SRC certmonger.c)
    add_library(certmonger MODULE ${PLUGIN_CERTMONGER_SRC})
    target_link_libraries(certmonger PRIVATE libmetric libutils LibSystemd::LibSystemd)
    set_target_properties(certmonger PROPERTIES PREFIX "")
    install(TARGETS certmonger DESTINATION ${CMAKE_INSTALL_LIBDIR}/ncollectd/)
    configure_file(ncollectd-certmonger.5 ncollectd-certmonger.5 @ONLY)
    install(FILES ${CMAKE_CURRENT_BINARY_DIR}/ncollectd-certmonger.5 DESTINATION ${CMAKE_INSTALL_MANDIR}/man5)
endif()
